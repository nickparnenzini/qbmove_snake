find_package(Protobuf REQUIRED)

set(PROTOBUF_IMPORT_DIRS)
foreach(ITR ${GAZEBO_INCLUDE_DIRS})
  if(ITR MATCHES ".*gazebo-[0-9.]+$")
    set(PROTOBUF_IMPORT_DIRS "${ITR}/gazebo/msgs/proto")
  endif()
endforeach()

set (msgs
  position_stiffness_request.proto 
  pos_current_echo_request.proto 
)
PROTOBUF_GENERATE_CPP(PROTO_SRCS PROTO_HDRS ${msgs})
add_library(position_stiffness_creator_msgs SHARED ${PROTO_SRCS})
target_link_libraries(position_stiffness_creator_msgs ${PROTOBUF_LIBRARY})

add_library(pos_current_echo_creator_msgs SHARED ${PROTO_SRCS})
target_link_libraries(pos_current_echo_creator_msgs ${PROTOBUF_LIBRARY})
